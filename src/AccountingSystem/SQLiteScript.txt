CREATE TABLE CUSTOMERS
(ID INTEGER PRIMARY KEY AUTOINCREMENT,
 NAME CHAR(40),
PHONE CHAR(40),
 ADDRESS CHAR(40));
 
CREATE TABLE ORDERS
(ID INTEGER PRIMARY KEY AUTOINCREMENT,
 CUSTOMER INTEGER,
 STATUS INTEGER,
 ORDER_DATE DATE,
 PRICE REAL,
 FOREIGN KEY(Customer) REFERENCES Customers(ID) on delete CASCADE);
 
CREATE TABLE PRODUCTS
(ID INTEGER PRIMARY KEY AUTOINCREMENT,
 NAME CHAR(40),
 PRICE REAL,
 IDX INTEGER default NULL, 
 ORDER_ID INTEGER default NULL,
 DATA DATE);
