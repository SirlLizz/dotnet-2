CREATE TABLE CUSTOMERS
(ID VARCHAR2(40),
 NAME VARCHAR2(40),
PHONE VARCHAR2(40),
 ADDRESS VARCHAR2(40));
 
CREATE TABLE ORDERS
(ID VARCHAR2(40),
 CUSTOMER VARCHAR2(40),
 STATUS NUMBER(38),
 PRICE NUMBER(38),
 PRODUCTS VARCHAR2(100));
 
CREATE TABLE PRODUCTS
(ID VARCHAR2(40),
 NAME VARCHAR2(40),
 PRICE NUMBER(38),
 DATA DATE);

ALTER TABLE CUSTOMERS
    ADD primary key (ID);
    
ALTER TABLE ORDERS
    ADD primary key(CUSTOMER);

ALTER TABLE Orders
    ADD FOREIGN KEY(Customer) REFERENCES Customers(Id) on delete CASCADE;
    
ALTER TABLE PRODUCTS
    ADD primary key (ID);
    
ALTER TABLE ORDERS
    ADD primary key(PRODUCTS);

ALTER TABLE Orders
    ADD FOREIGN KEY(PRODUCTS) REFERENCES PRODUCTS(Id) on delete CASCADE;
