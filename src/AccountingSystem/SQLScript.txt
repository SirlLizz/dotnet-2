CREATE TABLE CUSTOMERS
(ID number generated by default as identity,
 NAME VARCHAR2(40),
PHONE VARCHAR2(40),
 ADDRESS VARCHAR2(40));
 
CREATE TABLE ORDERS
(ID number generated by default as identity,
 CUSTOMER number,
 STATUS NUMBER(38),
 PRICE NUMBER(38));
 
CREATE TABLE PRODUCTS
(ID number generated by default as identity,
 NAME VARCHAR2(40),
 PRICE NUMBER(38),
 IDX NUMBER default NULL, 
 ORDER_ID NUMBER default NULL,
 DATA DATE);
